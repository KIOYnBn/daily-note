以下是基于论文描述的 **GCN + BiGRU + Attention** 药物分子编码器的 PyTorch 实现。代码包含：

-   利用 RDKit 从 SMILES 构建分子图，提取原子/键特征（遵循论文表 3-1）
-   自定义支持边特征的 GCN 层（实现公式 3-14）
-   多感受野特征收集 + BiGRU + 注意力融合（实现公式 3-15 至 3-20）
-   完整 DrugEncoder 模块，输出每个原子的最终嵌入

## 2. 原子与键特征提取（基于 RDKit）
根据论文表 3-1 定义特征函数。
```python
import torch
import numpy as np
from rdkit import Chem
from rdkit.Chem import rdchem

# 原子特征维度 51
ATOM_FEATURES = {
    'atomic_num': 33,          # [C, N, O, S, F, Si, P, Cl, Br, Mg, Na, Ca, Fe, As, Al, I, B, V, K, Sn, Ag, Pd, Co, Se, Ti, Zn, Li, Ge, Cu, Au, Ni, Mn, Other]
    'degree': 6,                # 共价键数量 0-5
    'formal_charge': 1,         # 形式电荷
    'num_radical_electrons': 1, # 未成对电子数
    'hybridization': 6,         # [SP, SP2, SP3, SP3D, SP3D2, Other]
    'is_aromatic': 1,           # 是否芳香
    'chirality': 3,             # [R, S, Other]
}

# 键特征维度 10
BOND_FEATURES = {
    'bond_type': 4,             # [Single, Double, Triple, Aromatic]
    'is_conjugated': 1,         # 是否共轭
    'is_in_ring': 1,            # 是否在环内
    'stereo': 4,                # [StereoNone, StereoAny, StereoZ, StereoE]
}

def one_hot_encoding(value, choices):
    """返回长度为 len(choices) 的 one-hot 向量"""
    encoding = [0] * len(choices)
    if value in choices:
        encoding[choices.index(value)] = 1
    return encoding

def get_atom_features(atom):
    features = []
    # 原子类型
    atomic_num = atom.GetAtomicNum()
    atomic_num_list = [5, 6, 7, 8, 9, 14, 15, 16, 17, 35, 53, 3, 11, 12, 19, 20, 25, 26, 27, 28, 29, 30, 34, 47, 48, 50, 53, 78, 79, 80, 82, 83, 0]  # 论文中的列表，这里简化
    # 实际使用时需要完整映射，此处简化为 33 维
    atomic_choices = list(range(1, 34))  # 伪代码，需按论文顺序
    features.extend(one_hot_encoding(atomic_num if atomic_num < 34 else 33, atomic_choices))

    # 共价键数量（度数）
    degree = atom.GetDegree()
    degree_choices = [0,1,2,3,4,5]
    features.extend(one_hot_encoding(degree, degree_choices))

    # 形式电荷
    formal_charge = atom.GetFormalCharge()
    features.append(formal_charge)

    # 未成对电子数
    num_radical = atom.GetNumRadicalElectrons()
    features.append(num_radical)

    # 手性类型
    chiral_tag = atom.GetChiralTag()
    chiral_choices = [rdchem.ChiralType.CHI_UNSPECIFIED,
                      rdchem.ChiralType.CHI_TETRAHEDRAL_CW,
                      rdchem.ChiralType.CHI_TETRAHEDRAL_CCW]
    chiral_idx = chiral_choices.index(chiral_tag) if chiral_tag in chiral_choices else 0
    features.extend(one_hot_encoding(chiral_idx, [0,1,2]))

    # 杂化类型
    hybrid = atom.GetHybridization()
    hybrid_choices = [rdchem.HybridizationType.SP,
                      rdchem.HybridizationType.SP2,
                      rdchem.HybridizationType.SP3,
                      rdchem.HybridizationType.SP3D,
                      rdchem.HybridizationType.SP3D2,
                      rdchem.HybridizationType.UNSPECIFIED]
    hybrid_idx = hybrid_choices.index(hybrid) if hybrid in hybrid_choices else 5
    features.extend(one_hot_encoding(hybrid_idx, list(range(6))))

    # 是否芳香
    features.append(1 if atom.GetIsAromatic() else 0)

    return features

def get_bond_features(bond):
    features = []
    # 键类型
    bond_type = bond.GetBondType()
    bond_choices = [rdchem.BondType.SINGLE,
                    rdchem.BondType.DOUBLE,
                    rdchem.BondType.TRIPLE,
                    rdchem.BondType.AROMATIC]
    bond_type_idx = bond_choices.index(bond_type) if bond_type in bond_choices else 0
    features.extend(one_hot_encoding(bond_type_idx, [0,1,2,3]))

    # 是否共轭
    features.append(1 if bond.GetIsConjugated() else 0)

    # 是否在环内
    features.append(1 if bond.IsInRing() else 0)

    # 立体构型
    stereo = bond.GetStereo()
    stereo_choices = [rdchem.BondStereo.STEREONONE,
                      rdchem.BondStereo.STEREOANY,
                      rdchem.BondStereo.STEREOZ,
                      rdchem.BondStereo.STEREOE]
    stereo_idx = stereo_choices.index(stereo) if stereo in stereo_choices else 0
    features.extend(one_hot_encoding(stereo_idx, [0,1,2,3]))

    return features
```

## 3. 支持边特征的 GCN 层（自定义）
论文公式 (3-14) 要求节点更新时同时聚合邻居节点特征和边特征。我们使用 PyG 的 `MessagePassing` 实现。
```python
import torch
import torch.nn as nn
import torch.nn.functional as F
from torch_geometric.nn import MessagePassing
from torch_geometric.utils import add_self_loops, degree

class EdgeGCNConv(MessagePassing):
    def __init__(self, in_channels, out_channels, edge_dim):
        super().__init__(aggr='add')  # 聚合方式为 sum
        self.lin_node = nn.Linear(in_channels, out_channels)
        self.lin_edge = nn.Linear(edge_dim, out_channels)
        self.root = nn.Linear(in_channels, out_channels)  # 自身节点变换

    def forward(self, x, edge_index, edge_attr):
        # x: [N, in_channels]
        # edge_index: [2, E]
        # edge_attr: [E, edge_dim]

        # 添加自环（可选，论文中可能隐含自环？这里不加，因为公式中似乎没有自环）
        # 但为了保留自身信息，公式3-14中有单独的 self node term，这里通过 root 处理
        x_node = self.lin_node(x)
        x_edge = self.lin_edge(edge_attr)

        # 开始消息传递
        out = self.propagate(edge_index, x=x_node, edge_attr=x_edge)

        # 加上自身节点变换
        out = out + self.root(x)

        return out

    def message(self, x_j, edge_attr):
        # x_j: 邻居节点特征 [E, out_channels]
        # edge_attr: 边特征 [E, out_channels]
        # 公式 3-14 中是将邻居节点和边特征相加
        return x_j + edge_attr

    def update(self, aggr_out):
        # aggr_out: [N, out_channels]
        return aggr_out
```

## 4. 多感受野收集 + BiGRU + Attention
```python
class BiGRUAttention(nn.Module):
    def __init__(self, input_dim, hidden_dim, num_layers):
        super().__init__()
        self.hidden_dim = hidden_dim
        self.num_layers = num_layers
        # 双向 GRU，输出维度 2*hidden_dim
        self.gru = nn.GRU(input_dim, hidden_dim, batch_first=True, bidirectional=True)
        # 注意力参数
        self.attn = nn.Linear(2 * hidden_dim, 1, bias=False)

    def forward(self, x):
        # x: [batch_size, num_nodes, num_layers, feat_dim]  (每个节点每层的特征)
        batch_size, num_nodes, L, feat_dim = x.shape
        x = x.view(batch_size * num_nodes, L, feat_dim)  # [B*N, L, feat_dim]

        # BiGRU
        gru_out, _ = self.gru(x)  # [B*N, L, 2*hidden_dim]

        # 注意力得分
        attn_weights = self.attn(gru_out).squeeze(-1)  # [B*N, L]
        attn_weights = F.softmax(attn_weights, dim=-1)  # [B*N, L]

        # 加权求和
        weighted = torch.sum(attn_weights.unsqueeze(-1) * gru_out, dim=1)  # [B*N, 2*hidden_dim]

        # 恢复形状
        weighted = weighted.view(batch_size, num_nodes, -1)  # [B, N, 2*hidden_dim]
        return weighted
```

## 5. 完整的药物编码器
```python
class DrugEncoder(nn.Module):
    def __init__(self, node_dim=51, edge_dim=10, hidden_dim=128, num_gcn_layers=8, gru_hidden=128):
        super().__init__()
        self.num_layers = num_gcn_layers
        self.hidden_dim = hidden_dim

        # 初始嵌入
        self.node_embed = nn.Linear(node_dim, hidden_dim)
        self.edge_embed = nn.Linear(edge_dim, hidden_dim)

        # 堆叠 GCN 层
        self.convs = nn.ModuleList()
        for _ in range(num_gcn_layers):
            self.convs.append(EdgeGCNConv(hidden_dim, hidden_dim, hidden_dim))

        # BiGRU + Attention
        self.gru_attn = BiGRUAttention(hidden_dim, gru_hidden, num_gcn_layers)

        # 最终线性变换到与蛋白质相同的维度 H（比如 256）
        self.fc_out = nn.Linear(2 * gru_hidden, 256)  # 2*gru_hidden = 256 if gru_hidden=128

    def forward(self, data):
        # data 包含 x, edge_index, edge_attr
        x = self.node_embed(data.x)           # [N, hidden_dim]
        edge_attr = self.edge_embed(data.edge_attr)  # [E, hidden_dim]

        # 保存每一层的输出
        layer_outputs = []
        for conv in self.convs:
            x = conv(x, data.edge_index, edge_attr)
            x = F.relu(x)
            layer_outputs.append(x)  # 每层输出 shape [N, hidden_dim]

        # 堆叠层输出 -> [num_layers, N, hidden_dim]
        stacked = torch.stack(layer_outputs, dim=0)  # [L, N, H]
        stacked = stacked.permute(1, 0, 2)          # [N, L, H]

        # 增加 batch 维度，假设 batch_size=1，但通常需要支持 batch
        # 实际使用时，data 是 PyG Batch 对象，需要处理成 batch 形式
        # 这里简单起见，假设 batch 已通过 PyG 的 Batch 类处理
        # 从 PyG Batch 中提取 batch 信息
        batch = data.batch if hasattr(data, 'batch') else torch.zeros(x.size(0), dtype=torch.long)
        unique_batches = batch.unique()
        batch_size = len(unique_batches)

        # 按 batch 分割节点
        node_per_graph = []
        for b in unique_batches:
            mask = (batch == b)
            node_per_graph.append(stacked[mask].unsqueeze(0))  # [1, N_b, L, H]

        # 对每个图分别处理，然后拼接
        final_outputs = []
        for i, node_feat in enumerate(node_per_graph):
            # node_feat: [1, N_i, L, H]
            out = self.gru_attn(node_feat)  # [1, N_i, 2*gru_hidden]
            out = self.fc_out(out)          # [1, N_i, 256]
            final_outputs.append(out.squeeze(0))  # [N_i, 256]

        # 如果需要全局图表示，可以池化，但论文后续使用共注意力，所以返回节点级特征
        return final_outputs  # 列表，每个元素是 [N_i, 256] 的节点嵌入
```

## 6. 数据预处理：SMILES → PyG Data
```python
from torch_geometric.data import Data

def smiles_to_graph(smiles):
    mol = Chem.MolFromSmiles(smiles)
    if mol is None:
        return None

    # 原子特征
    atom_features = []
    for atom in mol.GetAtoms():
        atom_features.append(get_atom_features(atom))
    x = torch.tensor(atom_features, dtype=torch.float)

    # 边特征
    edge_indices = []
    edge_attrs = []
    for bond in mol.GetBonds():
        i = bond.GetBeginAtomIdx()
        j = bond.GetEndAtomIdx()
        edge_indices += [[i, j], [j, i]]
        bond_feat = get_bond_features(bond)
        edge_attrs.append(bond_feat)
        edge_attrs.append(bond_feat)  # 无向图双向

    edge_index = torch.tensor(edge_indices, dtype=torch.long).t().contiguous()
    edge_attr = torch.tensor(edge_attrs, dtype=torch.float)

    data = Data(x=x, edge_index=edge_index, edge_attr=edge_attr)
    return data
```

## 8. 注意事项
-   特征提取中原子类型列表需要严格按照论文表 3-1 的 33 种元素顺序，上面代码仅为示例，实际使用时需完整映射。
-   论文中 GCN 聚合时同时使用了节点和边特征，自定义 `EdgeGCNConv` 实现了 `x_j + edge_attr`，符合公式 (3-14) 中 `W f_v + M f_{e}` 累加的思想（这里将变换后的特征相加）。
-   BiGRU-Att 中 `gru_hidden` 应与 GCN 隐藏维度一致（论文设为 128），输出维度为 256，经过线性层变为 256（与蛋白质特征对齐）。
-   论文中后续的 Co-attention 模块需要将药物节点特征与蛋白质残基特征进行交互，这里未实现，但可根据论文 3.5.2 节补充。

如果需要完整的 DTI 预测模型（含蛋白质编码器 + Co-attention + 分类器），可以进一步扩展。以上代码提供了核心的 GCN+BiGRU+Attention 药物编码器的完整复现。

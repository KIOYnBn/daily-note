def init_seeds(seed:int = 42):
	"Initialize random seeds for reproducibility."""
	def safe_set(name,func):
		try:func()
		except ImportError:pass
		except Exception as e:print(f"Error occuring when set {name} seed:{e}")
		return 0
		
	def set_torch():
		torch.manual_seed(seed)
		torch.cuda.manual_seed_all(seed)
		torch.backends.cudnn.deterministic = True
		torch.backends.cudnn.benchmark = False
		return 0
	
	safe_set("torch", set_torch)
	safe_set("numpy", lambda:np.random.seed(seed))
	safe_set("random", lambda:random.seed(seed))
	
	return 0

## Flash attn vs Original attn
```python
import torch
import torch.nn.functional as F
import time
import contextlib

# ---------- 准备数据 ----------
device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
dtype = torch.float16

batch_size = 1
num_heads = 8
seq_len = 2048      # 可改为 1024, 4096 等测试
head_dim = 64

torch.manual_seed(42)
query = torch.randn(batch_size, num_heads, seq_len, head_dim, device=device, dtype=dtype)
key   = torch.randn(batch_size, num_heads, seq_len, head_dim, device=device, dtype=dtype)
value = torch.randn(batch_size, num_heads, seq_len, head_dim, device=device, dtype=dtype)

print(f"Running on: {device}, dtype: {dtype}")
print(f"GPU: {torch.cuda.get_device_name(0) if device.type=='cuda' else 'CPU'}\n")

# ---------- 1. 手动普通 Attention（O(N^2) 显存） ----------
def manual_attention(q, k, v, causal=False):
    scale = q.size(-1) ** -0.5
    attn_weights = (q @ k.transpose(-2, -1)) * scale
    if causal:
        L, S = q.size(-2), k.size(-2)
        mask = torch.ones(L, S, dtype=torch.bool, device=q.device).tril()
        attn_weights.masked_fill_(mask.logical_not(), float("-inf"))
    attn_weights = torch.softmax(attn_weights, dim=-1)
    return attn_weights @ v

# ---------- 2. 强制数学后端的“普通 Attention”（SDPA API） ----------
def math_attention(q, k, v, causal=False):
    with torch.backends.cuda.sdp_kernel(enable_flash=False, enable_math=True, enable_mem_efficient=False):
        return F.scaled_dot_product_attention(q, k, v, is_causal=causal)

# ---------- 3. 自动 SDPA（实际跑的是 FlashAttention 或 Memory-Efficient） ----------
def auto_sdpa(q, k, v, causal=False):
    return F.scaled_dot_product_attention(q, k, v, is_causal=causal)

# ---------- 性能与显存测试 ----------
def benchmark(func, *args, warmup=3, repeat=10, name=""):
    # 预热
    for _ in range(warmup):
        func(*args)
    torch.cuda.synchronize()

    # 测时间
    start = time.time()
    for _ in range(repeat):
        func(*args)
    torch.cuda.synchronize()
    elapsed = (time.time() - start) / repeat

    # 测显存
    torch.cuda.reset_peak_memory_stats()
    torch.cuda.empty_cache()
    with torch.no_grad():
        out = func(*args)
    peak_mem = torch.cuda.max_memory_allocated() / 1024**3  # GB

    print(f"{name:40s} | time: {elapsed*1000:7.2f} ms | peak memory: {peak_mem:.2f} GB")
    return out, elapsed, peak_mem

print("===== Causal Attention Benchmark (seq_len={}) =====".format(seq_len))
out_manual, t_manual, m_manual = benchmark(manual_attention, query, key, value, True, name="Manual (naive)")
out_math,  t_math,  m_math  = benchmark(math_attention,  query, key, value, True, name="SDPA math backend (naive)")
out_auto,  t_auto,  m_auto  = benchmark(auto_sdpa,        query, key, value, True, name="SDPA auto (FlashAttn etc.)")

# ---------- 正确性验证 ----------
# 手动和强制数学后端的结果应该完全一致（或误差极小）
diff_manual_math = (out_manual - out_math).abs().max().item()
# 自动 SDPA 由于算法不同，允许稍有数值差异
diff_auto = (out_manual - out_auto).abs().max().item()
print(f"\nMax diff: manual vs. math backend = {diff_manual_math:.6f}")
print(f"Max diff: manual vs. auto SDPA     = {diff_auto:.6f}")
```

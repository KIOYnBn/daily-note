
# if语句
```c
#include <stdio.h>
int main(void) {
	int number = 10;
	if (number >= 10) {
		puts("number >= 10");
	}
	else if (0<=number<10) {
		puts("0 <number < 10");
	}
	else {
		puts("number为负数");
	}
	return 0;
}
```

# 简化if语句
```c
int main(void) {
	int score = 80;
	char grade;
	grade = (score > 90 ) ? 'A':
		(score >= 80) ? 'B':
		(score >= 70) ? 'C': 'D';
	return 0;
	
}

# switch-case语句
*  需要注意的是， switch里的(number)必须是整型
```c
int main(void) {
	int number = 8;
	switch (number ) {
		case 1:
			puts("one");
			break;
		case 2:
			puts("two");
			break;
		default:
			puts("over two");
			break;
	}
	return 0;
}


# 浮点数
* 类型
	* double： 默认， 8字节
	* float： 4字节
* c语言中， 默认的小数类型为double
```c
#include <stdio.h>

int main(void) {
	double a = 10.34;
	float b = 10.34F;
	printf("double: %.2f, float: %.2f", a, b);
	printf("科学计数法:%e, 十六进制: %a", b, b);
	return 0;
}
```

# 整数
* 当前工程中， 通常使用stdint.h中的int8_t, int16_t, int32_t, int64_t。而不用short, int, long
	* int32后， uint32需要加U, long类型加L
* 当前工程的格式化输出int， 通常使用inttypes.h的`PRId8`, `PRId16`等， 而不是%d等
* 整数类型： 
	* 十进制： 数字0~9开头
	* 八进制： 数字0开头
	* 十六进制： 数字0x开头， 不分区字母大小写
* 后缀： 
	* u： unsigned
	* l: long: 
	* ll: long long

|平台/模型|`int` 大小|`long` 大小|`long long` 大小|
|---|---|---|---|
|Windows (LLP64)|4 字节（32位）|4 字节（32位）|8 字节（64位）|
|Linux/macOS (LP64)|4 字节（32位）|8 字节（64位）|8 字节（64位）|
|16位系统（罕见）|2 字节（16位）|4 字节（32位）|8 字节（64位）|


```c
#include <stdio.h>

int main() {
	int dec_int = 28;
	short s_int = 28;
	int oct_int = 027;
	int x_int = 0x2A;
	unsigned dec_uint = 28u;
	long dec_long = 28l;
	long long dec_llong = 28ll;
	unsigned long long dec_ullong = 28ull;
	printf("i: %d, h:%hd, o: %#o, x: %#x, u: %u, l: %ld, ll: %lld, ull:%llu", dec_int, s_int, oct_int, x_int, dec_uint, dec_long, dec_llong, dec_ullong);
	return 0;
}
```

# 字符
* 字符： 由`''`包裹
* 字符串： 由`""`包裹， 表示一个字符序列
	* 多行字符串的末尾用`\`连接
	* 最长为509/2048个字符
* 多字节： 由一个或多个字节的序列构成的字符
	* 如`"hello"`占6字节（5个字母+\0）
	* 用于多语言： 如`"中"`在内存里可能2/3字节
* 宽字符： 宽度始终为16位（2字节）的字符代码
	* 占用空间大， 将文本变成数组
	* 适用于国际字符集unicode
	* 类型为： wchar_t
	* `L"hello"` -> `{L"h", L"e", L"l", L"l", L"o", 0}`
* 总结
```
`'a'` 是一个普通的 `char` 类型字符常量。
    
`L'a'` 是一个 `wchar_t` 类型的宽字符常量。
    
`"hello"` 是一个 `char` 数组（多字节字符串）。
    
`L"hello"` 是一个 `wchar_t` 数组（宽字符串）。
```

# 指针变量
* `指针具有最高权限， 直接指向实际存储位置`
* 指针： 指向变量的地址, 十六进制
	* 类型： 
		* 无类型指针: void *
		* int指针： int *
		* double指针： int *
	* 在变量后加`*`
* 取地址符： &
* 指针取值与修改： 指针变量前加*
* 指针输出： %p
* 野指针： 指向无效或已经释放的地址
* 空指针： 没有指向任何有效内存地址
	* int* ptr = NULL;
* 指针取值： 
	* 单个变量： 加`*`
	* 矩阵： 不需要加`*`——因为`ptr[n][m]会被解析为*(ptr+n*sizeof(var1)+m*sizeof(var2))`
```c
#include<stdio.h>

int main(void) {

	int num1 = 10;
	int *ptr_num1 = &num1;
	printf("the address fo the var num1 is %p\n", ptr_num1);
	*ptr_num1 = 100;
	printf("the result of num1 is %d\n", *ptr_num1);

	return 0;
}
```
## 指针在数组上的应用
```c
#include<stdio.h>
#include<inttypes.h>
#include<stdint.h>
typedef uint32_t u32;

int main(void) {

	u32 arrays[] = {0, 1, 2, 3, 4};
	
	// 赋值数组的第一个元素的地址
	u32* start_ptr = arrays; / 等同start_ptr = &arrays[0]
	printf("the start ptr of arrays is %p\n", start_ptr);
	
	// 指针的运算
	u32* tmp_ptr = NULL;
	size_t size_of_array = sizeof(arrays) / sizeof(u32);
	for (size_t index = 0; index < size_of_array; ++ index) {
		// 原理： ptr + value*sizeof(the type of var)
		tmp_ptr = start_ptr + index;
		printf("the value in %p is %" PRIu32 "\n", tmp_ptr, *(tmp_ptr));
	}
	
	// 指针的比较
	u32* end_ptr = &arrays[size_of_array-1];
	if (end_ptr > start_ptr) {
		puts("数组最后一个元素的地址大于第一个元素的地址\n");
	}			
	else {
		puts("数组最后一个元素的地址小于第一个元素的地址\n");
	}
	
	// 通过指针修改数组的值
	for (size_t index = 0; index < size_of_array; ++index) {
		*(start_ptr+index) += 1;
	}
	puts("修改后的数组为:\n{");
	for (size_t index = 0; index < size_of_array; ++index) {
		printf("%" PRIu32 ", ", *(start_ptr+index));
	}
	puts("}\n");

	return 0;
	
}
```
## 指针在多维数组上的应用
* `ptr[i][j] = ptr + i*第一维步长 + j*第二维步长`
```c
#include <stdio.h>

int main(void) {
	
	// matrix的变量类型为int (*)[3]
	int matrix[2][3] = {
		{0, 1, 2},
		{10, 11, 12}
	};
	
	int (*ptr)[3] = matrix;  // 等同于int (*ptr)[3] = matrix[0], 即类型为3*int
	printf("指针地址： %p\n", ptr);
	printf("数组起始地址%p\n", &matrix[0][0]);
	printf("数组第一维步长: %zu\n", sizeof(*ptr));
	printf("数组第二维步长: %zu\n", sizeof(*ptr[0]));
	
	// 获取数组每个值
	for (int i = 0; i < 2; ++i) {
	
		for (int j=0; j<3; ++j) {
		
			printf("%d ", ptr[i][j]);
		}
	}

	return 0;
}

```
## 指针在函数上的应用
* 传递地址
```c
#include <stdio.h>

void add_ten(int* value);

int main(void) {

	int num = 10;
	
	printf("num的值为: %d\n", num);
	add_ten(&num);
	printf("函数处理后num的值为: %d\n", num);
	
	return 0;
}

void add_ten(int* value) {
	*value += 10;
}
```

# 基本概念
* 字符： 由`''`包裹
* 字符串： 由`""`包裹， 表示一个字符序列
	* 多行字符串的末尾用`\`连接
	* 最长为509/2048个字符
* 多字节： 由一个或多个字节的序列构成的字符
	* 如`"hello"`占6字节（5个字母+\0）
	* 用于多语言： 如`"中"`在内存里可能2/3字节
* 宽字符： 宽度始终为16位（2字节）的字符代码
	* 占用空间大， 将文本变成数组
	* 适用于国际字符集unicode
	* 类型为： wchar_t
	* `L"hello"` -> `{L"h", L"e", L"l", L"l", L"o", 0}`
* 总结
```
`'a'` 是一个普通的 `char` 类型字符常量。
    
`L'a'` 是一个 `wchar_t` 类型的宽字符常量。
    
`"hello"` 是一个 `char` 数组（多字节字符串）。
    
`L"hello"` 是一个 `wchar_t` 数组（宽字符串）。
```

* 八进制转移序列： `\101`

# 字符串
- 字符串的初始化通常用strcpy_s， 这样更安全.
## 字符串的定义方式
```c
#include <stdio.h>

int main(void) {
	
	// --- A --- : 普通字符串
	// 方式一: 字符串数组
	char str[50] = "Hello World!";
	
	// 方式二: 指针字符串
	// 指针字符串定义后无法修改， 因此通常与const连用。因为修改指针字符串时会出现bug， 但是编辑器不报错
	const char* string = "Welcome!";
	puts(str);
	puts(string);
	printf("%s", str);
	
	// --- B --- : 宽字符串
	const wchar_t* wstr = L"世界";
	printf("%ls", wstr);
	return 0;
}
```
## 字符串的复制
- strcpy_s(目标字符串， 目标字符串大小， 被复制字符串)
- strncpy_s(目标字符串， 目标字符串大小， 被复制字符串， 最大复制长度)
- Note： 二者均返回一个errno_t类型的变量， 用于判断是否复制成功
```c
#include <stdio.h>
#include <string.h>

int main(void) {
	const char* str = "Hello World";
	char dest[50];
	char destN[50];
	
	errno_t result = strcpy_s(dest, sizeof(dest), str);
	if (result == 0) {
		puts(dest);
	}
	else {
		puts("Error occurred while copying string");
	}
	
	size_t max_copy = 5;
	errno_t resultN = strncpy_s(destN, sizeof(destN), str, max_copy);
	if (resultN == 0) {
		puts(destN);
	}
	else {
		puts("Error occurred while copying string");
	}


	return 0;
}
```
## 字符串拼接
- strcat_s(目标字符串， 目标字符串大小， 被拼接字符串)
- strncat_s(目标字符串， 目标字符串大小， 被拼接字符串， 最大拼接数)
- Note: 二者均返回一个int类型，为0时说明拼接成功
```c
#include <stdio.h>
#include <string.h>

int main(void) {

	const char* string = "Hello World";
	char dest[50];
	char destN[50];
	strcpy_s(dest, sizeof(dest), "Welcome");
	strcpy_s(destN, sizeof(dest), "Welcome");
	
	int result = strcat_s(dest, sizeof(dest), string);
	if (result == 0) {
		puts(dest);
	}
	else {
		puts("Error occurred while cating string");
	}
	
	int max_cat = 5;
	int resultN = strncat_s(destN, sizeof(destN), string, max_cat);
	if (resultN == 0) {
		puts(destN);
	}
	else {
		puts("Error occurred while cating string");
	}

	return 0;
}
```

## 字符串分割
- strtok_s： 根据字符串分隔字符串
```c
#include <stdio.h>
#include <string.h>

int main(void) {

	char str[] = "Hello World";
	char delim[] = " ";
	char* token;
	char* context = {0};
	token = strtok_s(str, delim, &context);
	while (token != NULL) {
		puts(token);
		token = strtok_s(NULL, delim, &context);
	}
	return 0;
}
```
## 字符比对
- strcmp： 判断两个字符串是不是相同
- strncmp： 判断字符串前几个字符是否相同
```c
#include <stdio.h>
#include <string.h>

int main(void) {
	const char* str1 = "Hello World";
	const char* str2 = "Hello TOM";
	int max_match = 5;
	
	if (strcmp(str1, str2) == 0) {
		puts("str1 matches str2");
	}
	else {
		puts("str1 does not match str2");
	}
	if (strncmp(str1, str2, max_match) == 0) {
		puts("The first five characters of string 1 match those of string 2");
	}
	else {
		puts("The first five characters of string 1 do not match those of string 2");
	}


	return 0;
}
```
## 字符串中查找字符
- strchr： 从字符串开头起始寻找字符， 返回指针
- strrchr：  从字符串结尾起始寻找字符， 返回指针
```c
#include <stdio.h>
#include <string.h>

int main(void) {
	char str[] = "Hello World";
	char* ptr = str;
	char target = 'o';

	char* target_pos = strchr(str, target);
	char* target_rev_pos = strrchr(str, target);
	printf("target_pos is %td", target_pos-ptr);
	printf("target_rev_pos is %td", target_rev_pos-ptr);

	return 0;
}
```
## 字符串中查找字符串
- strstr： 目标字符串在寻找字符串中的指针
- strspn: 两个字符串从开头起始， 连续相同相同的字符个数
- strcspn： 在一个字符串中，从头开始查找另一个字符串中的任意字符第一次出现的位置，并返回这个位置之前的字符个数
```c
#include <stdio.h>
#include <string.h>

int main(void) {
	char str[] = "Hello World";
	char* str_ptr = str;
	
	// strstr
	char target[] = "World";
	char* target_ptr = strstr(str, target);
	printf("%td\n", target_ptr-str_ptr);
	
	//strspn
	size_t match_len = strspn(str, target);
	printf("%zu\n", match_len);
	
	//strcspn
	size_t match_clen = strcspn(str, target);
	printf("%zu\n", match_clen);
	
	//strcspn用于判断文件名是否合法
	char input[] = "filename.txt";
	char invalid_chars[] = "/\\:*?\"<>|";
	if (strcspn(input, invalid_chars) < strlen(input)) {
		puts("the file name includes invalid characters\n");
	}
	else {
		puts("the file name is valid");
	}
	return 0;
}
```

```c
#include <stdio.h>
#include <string.h>

#define MAX_WORD_SIZE 100
#define MAX_SENTENCE_SIZE 1000
#define DELIMITER " ,.!?;:\n"	

void replace_word(
	const char* origin_sentence,
	const char* repalced_word, 
	const char* new_word, char* result);

int main(void) {
	char sentence[MAX_SENTENCE_SIZE] = "This is a simple sentence simple";
	const char* replaced_word = "simple";
	const char* new_word = "complex";
	char result[MAX_SENTENCE_SIZE] = { 0 };
	replace_word(sentence, replaced_word, new_word, result);
	puts(result);

	return 0;
}

void replace_word(
	const char* origin_sentence,
	const char* replaced_word, 
	const char* new_word,
	char* result) {


	char buffer[MAX_SENTENCE_SIZE] = { 0 };
	const char* temp = NULL;
	const char* pos = origin_sentence;
	size_t word_len;
	
	while ((temp = strstr(pos, replaced_word)) != NULL) {
		word_len = strlen(new_word);
		strncat_s(buffer, MAX_SENTENCE_SIZE, pos, temp - pos);
		strcat_s(buffer, MAX_SENTENCE_SIZE, new_word);
		pos = temp + word_len - 1;
	}
	strcat_s(buffer, MAX_SENTENCE_SIZE, pos);
	strcpy_s(result, MAX_SENTENCE_SIZE, buffer);
}
```
## 特定字符的个数
```c
#include <stdio.h>
#include <string.h>

int counter(const char* sentence, const char target);

int main(void) {
	const char* sentence = "Hello, World";
	const char target_chr = 'o';
	int count = counter(sentence, target_chr);
	printf("%d", count);
	return 0;
}

int counter(const char* sentence, const char target) {
	int count = 0;
	while (*sentence) {
		if (*sentence == target) {
			++count;
		}
		++sentence;
	}
	return count;
}
```

## 提取每个单词
```c
#include <stdio.h>
#include <string.h>

#define DELMS " .!?\n"

void extract(const char* sentence, int* count_ptr, char(*dict_ptr)[20]);

int main(void) {

	const char* sentence = "Hello World";
	int count = 0;
	char dict[100][20] = { 0 };

	extract(sentence, &count, dict);

	for (int index = 0; index < count; index++) {
		printf("%s\n", dict[index]);
	}

	return 0;
}
void extract(const char* sentence, int* count_ptr, char(*dict_ptr)[20]) {
	char* context = NULL;
	char buffer[100] = { 0 };
	strcpy_s(buffer, sizeof(buffer), sentence);
	char* token = strtok_s(buffer, DELMS, &context);
	while (token != NULL) {
		strcpy_s(dict_ptr[*count_ptr], 20, token);
		(*count_ptr)++;
		token = strtok_s(NULL, DELMS, &context);
	}
}
```

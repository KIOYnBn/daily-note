* `指针具有最高权限， 直接指向实际存储位置`
* 指针： 指向变量的地址, 十六进制
	* 类型： 
		* 无类型指针: void *
		* int指针： int *
		* double指针： int *
	* 在变量后加`*`
* 取地址符： &
* 指针取值与修改： 指针变量前加*
* 指针输出： %p
* 野指针： 指向无效或已经释放的地址
* 空指针： 没有指向任何有效内存地址
	* int* ptr = NULL;
* 指针取值： 
	* 单个变量： 加`*`
	* 矩阵： 不需要加`*`——因为`ptr[n][m]会被解析为*(ptr+n*sizeof(var1)+m*sizeof(var2))`
* 指针差输出： `td`
```c
#include<stdio.h>

int main(void) {

	int num1 = 10;
	int *ptr_num1 = &num1;
	printf("the address fo the var num1 is %p\n", ptr_num1);
	*ptr_num1 = 100;
	printf("the result of num1 is %d\n", *ptr_num1);
	
	int num2 = 50;
	int* ptr_num2 = &num2;
	printf("%td", ptr_num2-ptr_num1);

	return 0;
}
```
## 指针在数组上的应用
```c
#include<stdio.h>
#include<inttypes.h>
#include<stdint.h>
typedef uint32_t u32;

int main(void) {

	u32 arrays[] = {0, 1, 2, 3, 4};
	
	// 赋值数组的第一个元素的地址
	u32* start_ptr = arrays; / 等同start_ptr = &arrays[0]
	printf("the start ptr of arrays is %p\n", start_ptr);
	
	// 指针的运算
	u32* tmp_ptr = NULL;
	size_t size_of_array = sizeof(arrays) / sizeof(u32);
	for (size_t index = 0; index < size_of_array; ++ index) {
		// 原理： ptr + value*sizeof(the type of var)
		tmp_ptr = start_ptr + index;
		printf("the value in %p is %" PRIu32 "\n", tmp_ptr, *(tmp_ptr));
	}
	
	// 指针的比较
	u32* end_ptr = &arrays[size_of_array-1];
	if (end_ptr > start_ptr) {
		puts("数组最后一个元素的地址大于第一个元素的地址\n");
	}			
	else {
		puts("数组最后一个元素的地址小于第一个元素的地址\n");
	}
	
	// 通过指针修改数组的值
	for (size_t index = 0; index < size_of_array; ++index) {
		*(start_ptr+index) += 1;
	}
	puts("修改后的数组为:\n{");
	for (size_t index = 0; index < size_of_array; ++index) {
		printf("%" PRIu32 ", ", *(start_ptr+index));
	}
	puts("}\n");

	return 0;
	
}
```
## 指针在多维数组上的应用
* `ptr[i][j] = ptr + i*第一维步长 + j*第二维步长`
```c
#include <stdio.h>

int main(void) {
	
	// matrix的变量类型为int (*)[3]
	int matrix[2][3] = {
		{0, 1, 2},
		{10, 11, 12}
	};
	
	int (*ptr)[3] = matrix;  // 等同于int (*ptr)[3] = matrix[0], 即类型为3*int
	printf("指针地址： %p\n", ptr);
	printf("数组起始地址%p\n", &matrix[0][0]);
	printf("数组第一维步长: %zu\n", sizeof(*ptr));
	printf("数组第二维步长: %zu\n", sizeof(*ptr[0]));
	
	// 获取数组每个值
	for (int i = 0; i < 2; ++i) {
	
		for (int j=0; j<3; ++j) {
		
			printf("%d ", ptr[i][j]);
		}
	}

	return 0;
}

```
## 指针在函数上的应用
* 传递地址
```c
#include <stdio.h>

void add_ten(int* value);

int main(void) {

	int num = 10;
	
	printf("num的值为: %d\n", num);
	add_ten(&num);
	printf("函数处理后num的值为: %d\n", num);
	
	return 0;
}

void add_ten(int* value) {
	*value += 10;

# 多级指针
## 用途
- 核心： 指针用于修改其指向的对象的值
	- 故， 一级指针用来修改其指向对象的值, 如player.name
	- 二级指针用来修改其指向的一级指针的值
	- 在`示例`中， 一级指针指向玩家列表， 由于一直有新玩家假如， 故玩家列表在不断变化， 故一级指针（列表的首元素的指针）在不断变化
## 用途示例
```c
#include <stdio.h>
#include <errno.h>
#include <stdlib.h>

typedef struct {
	char name[20];
	int id;
} Player;

void update_player_list(Player** players, Player* new_player, int* player_count);

void free_players(Player* players, int* player_count);
int main(void) {
	Player* players = NULL;
	int player_count = 0;

	Player p1 = { "Alice", 1 };
	update_player_list(&players, &p1, &player_count);
	Player p2 = { "Bob", 2 };
	update_player_list(&players, &p2, &player_count);
	printf("%s, %d\n", players[0].name, players[0].id);
	free_players(players, &player_count);


	return 0;
}

void update_player_list(Player** players, Player* new_player, int* player_count) {
	if (new_player == NULL) {
		perror("New player is NULL");
		exit(EXIT_FAILURE);
	}
	*player_count += 1;

	Player* temp = (Player*)realloc(*players, sizeof(Player)*(*player_count));
	if (temp == NULL) {
		perror("Failed to allocate memory");
		exit(EXIT_FAILURE);
	}
	*players = temp;
	(*players)[*player_count - 1] = *new_player;
}

void free_players(Player* players, int* player_count) {
	if (players != NULL) {
		free(players);
		*player_count = 0;
		puts("Player list freed successfully.");
	}


}

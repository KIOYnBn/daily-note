# 结构体变量
- 结构体也可以作为`函数参数`
- 结构体可`嵌套`
```c
#include <stdio.h>

typedef struct Person {
	char name[10];
	int height;
} Person;

// 匿名结构体
typedef struct {
	int x;
	int y;
} Point;

int main(void) {
	Person my = { "ljy", 180 };
	printf("%s 's height is %d\n", my.name, my.height);
	Person* my_ptr = &my;
	printf("%p\n", my_ptr);
	printf("%s\n", my_ptr->name);
	puts("after a year\n");
	my_ptr->height = 190;
	printf("%s 's height is %d\n", my.name, my.height);
	

	return 0;
}
```
## 结构体安全初始化
- 通过在函数中初始化
	- 作用： 具有安全作用（初始化指针是函数变量， 其指针在函数结束后被销毁）
```c
#include <stdio.h>

typedef struct Point {
	int x;
	int y;
} Point;

Point generatePoint(void);

int main(void) {

	Point p = generatePoint();
	printf("x: %d\ty: %d", p.x, p.y);
	return 0;
}

Point generatePoint(void) {
	Point p = {10, 10};
	return p;
}

```
## 结构体数组
```c
#include <stdio.h>

typedef struct {
	int x;
	int y;
} Point;


int main(void) {
	Point p[2] = {
		{1, 2}, {3, 4}
	};
	Point* ptr_p = p;
	
	for (int index = 0; index < 2; ++index) {
		printf("x: %d\ty: %d\n", ptr_p[index].x, ptr_p[index].y);
	
	}
	return 0;
}
```

# 联合
* 多种数据类型共同同一片内存空间， 每个时间段只能存在一种数据类型
```c
#include <stdio.h>

typedef union {
	int int_value;
	float float_value;
	char* string_value;
} Data;

int main(void) {
	Data data;
	data.int_value = 8;
	printf("%d\n", data.int_value);
	data.float_value = 0.1;
	printf("%.2f\n", data.float_value);

	return 0;
}
```

# Struct, Eum, Union的联合使用
```c
#include <stdio.h>

typedef union {
	int int_value;
	float float_value;
	char* string_value;
} DataValue;

typedef enum {
	INT, FLOAT, STRING
} DataType;

typedef struct {
	DataType type;
	DataValue value;
} Data;

void printData(Data* data);

int main(void) {
	Data data = { INT, {.int_value = 10} };
	printData(&data);
	return 0;
}

void printData(Data* data) {
	switch (data->type) {
	case INT: printf("Integer: %d\n", data->value.int_value);break;
	case FLOAT: printf("Float: %2.f\n", data->value.float_value);break;
	case STRING: printf("String: %s\n", data->value.string_value);break;
	}
}
```

#include <stdio.h>
#include <stdlib.h>

typedef struct {
	int data;
	struct List* next;
} List;

void addHead(List** head, int data);
void printList(List* head);

int main(void) {
	List* head = NULL;
	addHead(&head, 1);
	addHead(&head, 2);
	addHead(&head, 3);
	printList(head);
	free(head);
	return 0;
}

void addHead(List** head, int data) {
	List* tmp = (List*)malloc(sizeof(List));
	if (tmp == NULL) {
		perror("Failing to add new head");
		exit(EXIT_FAILURE);
	}
	tmp->data = data;
	tmp->next = *head;
	*head = tmp;
}

void printList(List* head) {
	List* temp = head;
	while (temp != NULL) {
		printf("%d", temp->data);
		temp = temp->next;
	}

}

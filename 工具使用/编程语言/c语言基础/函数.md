# 卫语句
* 核心思想： 在函数 / 代码块的开头，提前检查异常 / 边界条件，一旦满足就立即`返回 / 退出`，避免嵌套的条件判断
* 通常用于简化if逻辑判断
* 常规if示例
```
// 计算两个正整数的和（仅当num1、num2都>0时有效） 
int add_positive(int num1, int num2) { 
	int result = 0; 
	if (num1 > 0) { 
		// 第一层嵌套 
		if (num2 > 0) { 
			// 第二层嵌套 
			result = num1 + num2; 
		} 
		else { 
			printf("num2必须是正数！\n"); 
		} 
	} 
	else { 
		printf("num1必须是正数！\n"); } return result; 
	}
```

* 卫语句示例：
```
int add_positive(int num1, int num2) { 
	// 卫语句1：检查num1，不满足则直接返回 
	if (num1 <= 0) { 
		printf("num1必须是正数！\n"); 
		return 0; // 异常返回值（也可返回-1/EXIT_FAILURE等） 
	} 
	// 卫语句2：检查num2，不满足则直接返回 
	if (num2 <= 0) { 
		printf("num2必须是正数！\n"); 
		return 0; 
	} 
	// 所有异常条件都被“守卫”挡掉，剩下的就是正常逻辑 
	return num1 + num2; 
}
```

# 递归

* 本质： 函数嵌套

# 可变参数
```c
#include <stdio.h>
#include <stdarg.h>

int sum(int arg_count, ...) {
	va_list args;
	va_start(args, arg_count);
	int sum = 0;
	for (size_t i = 0; i < arg_count; ++i) {
		sum += va_arg(args, int);
	}
	va_end(args);
	return sum;
}

int main(void) {
	int num[] = { 1, 2, 3, 4, 5 };
	printf("the sum of 1, 2, 3, 4, 5 is %d\n", sum(5, num[0], num[1], num[2], num[3], num[4]));

	return 0;
}
```

## 基本内容
- FILE* 文件流
- fopen_s： 打开文件
- fclose： 文件关闭
- perror与EXIT_FAILURE: 捕获错误（stdlib.h, errno.h)
## 读取
- 关键函数
	- fgets: 逐行读取
	- memset： 释放内存(string.h)
	- fgetc： 逐字符读取
	- putchar： 打印字符
	- fscanf_s： 格式化读取
```c
#include <stdio.h>
#include <errno.h>
#include <stdlib.h>
#include <string.h>
int main(void) {
	FILE* file_stream = NULL;

	errno_t err = fopen_s(&file_stream, "./Text.txt", "r");

	char buffer[50];
	if (err != 0 || file_stream == NULL) {

		perror("error opening file");
		return EXIT_FAILURE;
	}

	// 逐行读取
	while ((fgets(buffer, sizeof(buffer), file_stream)) != NULL) {
		puts(buffer);
	}

	memset(buffer, 0, sizeof(buffer));
	rewind(file_stream);
	puts("");
	
	// 逐字符读取
	char ch;
	while ((ch = fgetc(file_stream)) != EOF) {
		putchar(ch);
	}
	
	// 格式化读取
	char str[100] = { 0 };
	int num = 0;
	if (fscanf_s(file_stream, " %s", str, (unsigned int)sizeof(str)) == 1) {
		printf("%s\n", str);
	}
	if (fscanf_s(file_stream, " %d", &num)) {
		printf("%d\n", num);
	}
	fclose(file_stream);
	
	fclose(file_stream);
	
	return 0;
}
```
## 写入
- fputs: 写入字符串
- fputc: 写入字符
- fprintf_s: 格式化写入
```c

#include <stdio.h>
#include <stdlib.h>

int main(void) {

	FILE* file_stream = NULL;

	errno_t err = fopen_s(&file_stream, "./Text.txt", "w");

		perror("Opening Error");
		return EXIT_FAILURE;
	}
	fputs("Hello World\n", file_stream);

	fprintf_s(file_stream, " This is a number which value is %d", num);

	return 0;

```
## 文件指针位置
- ftell: 当前指针位置
- fseek： 调整指针位置
- rewind： 回到文件指针开头
```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

int main(void) {

	FILE* file_stream = NULL;
	errno_t err = fopen_s(&file_stream, "./Text.txt", "r");
	if (err) {
		puts("Opening Error");
		return EXIT_FAILURE;
	}
	long position = ftell(file_stream);
	char buffer[100] = { 0 };
	printf("Current Position is %ld\n", position);
	if (fgets(buffer, sizeof(buffer), file_stream) != EOF) {
		puts(buffer);
		position = ftell(file_stream);
		printf("After using the function, called fgets, the position is %ld\n", position);
	}
	fseek(file_stream, 0, SEEK_SET);
	printf("Finally, the position is %ld\n", ftell(file_stream));
	fclose(file_stream);

	return 0;
}
```
## 错误处理
- 关键函数
	- ferror: 读取文件时的错误抓取
	- feo; 文件到达末尾
	- clearerr： 清除错误状态
	- exit： 退出程序
```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

int main(void) {

	FILE* file_stream = NULL;
	errno_t err = fopen_s(&file_stream, "./Text.txt", "r");
	
	// 文件打开错误检验
	if (err) {
		char error_msg[256];
		strerror_s(error_msg, sizeof(error_msg), errno);
		fprintf(stderr, "Failing to Open File: %s\n", error_msg);
		exit(EXIT_FAILURE);
	}
	char buffer[200] = { 0 };
	if (fgets(buffer, sizeof(buffer), file_stream) != NULL) {
		puts(buffer);
	}
	
	// 文件读取错误检验
	if (ferror(file_stream)) {
		puts("appearing error when reading file");
		clearerr(file_stream);
	}
	
	// 文件结尾错误检验
	if (feof(file_stream)) {
		puts("Reaching the end of file");
	}

	// 文件关闭错误检验
	if (fclose(file_stream) != 0) {
		perror("Closing File Failture");
		return EXIT_FAILURE;
	}

	return 0;

}
```

## 修改文件内容
```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <errno.h>

typedef enum {
	OPEN_FILE,
	READ_FILE,
	CLOSE_FILE,
} FILE_TIME;

typedef union {
	FILE * file_stream;
	errno_t err;
} FILE_ERR;

typedef struct {
	FILE_TIME period;
	FILE_ERR value;
} CHECK_FILE;

const char temp_file_name[] = "temp.log";

void checkpoint_file(CHECK_FILE* description);

void amend_file(char* file_path, char* origin_context, char* new_context);

int main(void) {
	char* file_path = "./Text.txt";
	char* origin_context = "Hello";
	char* new_context = "World";
	amend_file(file_path, origin_context, new_context);

	return 0;
}

void checkpoint_file(CHECK_FILE* description) {
	if (description == NULL) {
		fprintf(stderr, "Invalid description pointer\n");
		exit(EXIT_FAILURE);
	}
	switch (description->period) {
		case (OPEN_FILE): {
			errno_t err = description->value.err;
			if (err != 0) {
				char buffer[256];
				strerror_s(buffer, sizeof(buffer), err);
				fprintf(stderr, "Failing to opening file: %s\n", buffer);
				exit(EXIT_FAILURE);
			}
			break;
		}
		case (READ_FILE): {
			if (ferror(description->value.file_stream)) {
				perror("Reading file Error");
				exit(EXIT_FAILURE);
			}
			break;
		}
		case (CLOSE_FILE): {
			if (fclose(description->value.file_stream)) {
				perror("Error occured while closing file");
				exit(EXIT_FAILURE);
			}
			break;
		}
		default:
			fprintf(stderr, "Unknown file operation period\n");
			exit(EXIT_FAILURE);
	}
}

void amend_file(char* file_path, char* origin_context, char* new_context) {
	CHECK_FILE description;

	FILE* file_stream = NULL;
	FILE* temp_file_stream = NULL;

	errno_t err = fopen_s(&temp_file_stream, temp_file_name, "w");
	description.period = OPEN_FILE;
	description.value.err = err;
	checkpoint_file(&description);

	err = fopen_s(&file_stream, file_path, "r");
	description.value.err = err;
	checkpoint_file(&description);

	char buffer[1024] = { 0 };
	while (fgets(buffer, sizeof(buffer), file_stream) != NULL) {
		size_t len = strlen(buffer);
		if (len > 0 && buffer[len - 1] == '\n') {
			buffer[len - 1] = '\0';
		}
		if (strstr(buffer, origin_context) != NULL) {
			fprintf(temp_file_stream, "%s\n", new_context);
		}
		else {
			fprintf(temp_file_stream, "%s\n", buffer);
		}		
	}

	description.period = READ_FILE;
	description.value.file_stream = file_stream;
	checkpoint_file(&description);

	if (feof(file_stream)) {
		puts("Reaching the end of this file");
		clearerr(file_stream);
	}

	description.period = CLOSE_FILE;
	checkpoint_file(&description);
	description.value.file_stream = temp_file_stream;
	checkpoint_file(&description);
	remove(file_path);
	if (rename(temp_file_name, file_path) != 0) {
		perror("Failed to replace original file");
		exit(EXIT_FAILURE);
	}
	
	file_stream = NULL;
	temp_file_stream = NULL;
	puts("File amended successfully");

}
```

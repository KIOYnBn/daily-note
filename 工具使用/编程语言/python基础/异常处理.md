# 异常捕捉流程
```python
try:
	# 可能出错的代码
	pass
except:
	# 异常发生时的代码
	pass
else:
	# try正常运行时进行的
	pass
finally:
	# 无论try是否运行都执行
	pass
	
try:
	# 可能出错的代码
	pass
except ZeroDivisionError:
	pass
except ValueError:
	pass
	
try:
	pass
except (ZeroDivisionError, ValueError) as e:
	pass


```

# 自定义异常
```python

class ValidationError(Exception):
    def __init__(self, field, message, error_code=None):
        self.field = field
        self.error_code = error_code
        super().__init__(f"{field}: {message}")

def register_user(username, email):
    if len(username) < 3:
        raise ValidationError("username", "长度至少3个字符", error_code=1001)
    if "@" not in email:
        raise ValidationError("email", "邮箱格式不正确", error_code=1002)

try:
    register_user("ab", "test")
except ValidationError as e:
    print(f"字段 '{e.field}' 错误: {e} (错误码: {e.error_code})")
    # 可以向前端返回更结构化的错误信息
```

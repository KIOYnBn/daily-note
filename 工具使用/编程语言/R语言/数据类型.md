## 数据框
```R
# 数据框
name <- c("小红", "小明", "小李")
height <- c(155, 161, 196)
weight <- c(90, 112, 113)
information <- data.frame(name, height, weight)

# 数据提取
information[2,3] # [行， 列]
information["name"]
information[c("name", "height")]
information$height

# 数据框融合
data.frame()
d1 <- data.frame(
  x1 = c("a", "b", "c", "d"),
  y1 = c(1, 2, 3, 4)
)
d2 <- data.frame(
  x2 = c(11, 12, 13),
  y2 = c("a", "b", "c")
)
d3 = merge(d1, d2, by.x="x1", by.y = "y2") # 合并x1, y2中共有的部分， 非共有部分数据直接删除
a1 = data.frame(
  x1 = c(1, 2, 3),
  x2 = c("b", "c", "d")
)
a2 = data.frame(
  x1 = c(10, 7, 12),
  x2 = c("k1", "k2", "k3")
)
a3 = cbind(a1, a2)
a4 = rbind(a1, a2)

# 修改数据
fix(d1) # 在原变量修改
d2_copy = edit(d2) # 赋值给新变量
d1$x1[d1$x1 == "a"] <- "a1"
colnames(d1)[1] <- "x"
d1_copy = sub(1, 5, d1$y1) # 1替换成5
event <- "Don't give up"
event_1 <- strsplit(event, " ") # 按照空格分隔

# 缺失值处理
x <- c(1, 1, 1, NA)
is.na(x) # 检测是否有缺失值
na.rm(x) # 删除缺失值
x_copy = na.omit(x) # 删除具有缺失值的那一行

```
